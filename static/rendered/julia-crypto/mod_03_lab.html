::: brief id=theory title=Theory: Chinese Remainder Theorem Given $x \equiv a_i \pmod{n_i}$ with pairwise coprime moduli, CRT constructs the unique solution modulo $N = \prod n_i$:

$$x = \sum a_i \cdot N_i \cdot \text{invmod}(N_i, n_i) \pmod N$$

where $N_i = N / n_i$. :::

::: blueprint id=crt_impl lang=julia

function crt_two(r1, m1, r2, m2)
    N = m1 * m2
    N1 = N ÷ m1
    N2 = N ÷ m2
    x = (r1 * N1 * invmod(N1, m1) + r2 * N2 * invmod(N2, m2)) % N
    return x
end

println(crt_two(2, 3, 3, 5))  # x = 8
println(8 % 3, 8 % 5)
