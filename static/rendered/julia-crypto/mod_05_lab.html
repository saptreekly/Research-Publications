function sieve(n)
    is_prime = fill(true, n + 1)
    is_prime[1] = false

    for i in 2:n
        if is_prime[i] && i * i <= n
            for j in i*i:i:n
                is_prime[j] = false
            end
        end
    end

    return [i for i in 1:n if is_prime[i]]
end

function is_prime(n)
    n >= 2 && n in sieve(n)
end

println(sieve(30))
println(is_prime(17), is_prime(10))
