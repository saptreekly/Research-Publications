p, q = 61, 53
n = p * q
e = 7
d = invmod(e, (p - 1) * (q - 1))

m = 42
c = powermod(m, e, n)
m_decrypted = powermod(c, d, n)

println("Message:    $m")
println("Ciphertext: $c")
println("Decrypted:  $m_decrypted")
